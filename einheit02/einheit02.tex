\begin{frame}[fragile]{Bezeichner}
\begin{itemize}
\item \alert{Bezeichner} sind Namen, wie z.B. $x$ oder $f$. Sie können
im mathematischen Kontext sowohl Variablen als auch Unbestimmte repräsentieren.
\item Bezeichner sind aus Buchstaben, Ziffern und
Unterstrich \_ zusammengesetzt.
\item Sage unterscheidet zwischen Groß- und Kleinschreibung.
\item Bezeichner dürfen nicht mit einer Ziffer beginnen
\end{itemize}
Beispiele für Bezeichner
\begin{itemize}
\item zulässige Bezeichner:
\verb+x+, \verb+f+, \verb+x23+, \verb+_x_1+
\item unzulässige Bezeichner:
\verb+12x+, \verb+p~+, \verb+x>y+, \verb+Das System+
\end{itemize}
\end{frame}


\begin{frame}[fragile]{Wert eines Bezeichners}
\begin{itemize}
\item Der \alert{Wert} eines Bezeichners  ist ein \alert{Objekt} eines bestimmten
\alert{Datentyps}.
\item Ein \alert{Datentyp} ist durch seine Eigenschaften gegeben. \\
Beispiel: Natürliche Zahlen, rationale Zahlen, Bezeichner, Zeichenketten, \ldots  
\item Ein \alert{Objekt} ist eine Instanz (Einheit) eines Datentyps.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Zuweisungsoperator $:=$}
\begin{itemize}
\item Die Operation {\color{blue} \verb+bez=wert+} weist dem Bezeichner
\verb+bez+ den Wert \verb+wert+ zu.  
\item Beispiele: {\color{blue} \verb+N=5+}, {\color{blue} \verb+f = x^2-3*x-18+}
\item Rückgabeparameter ist die rechte Seite (Eine Ausgabe erfolgt jedoch normalerweise nicht)
\item Warnung: Unterscheiden Sie  stets zwischen dem Zuweisungsoperator {\color{blue} $=$} und
dem logischen Operator {\color{blue} $==$}.   
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Verbinden von Ausdrücken}
Ausdrücke können beliebig addiert, subtrahiert, multipliziert und
dividiert werden. 
\begin{itemize}
\item Definition
\begin{sage}
>> var('x,y'); f = x*x+3*x+y; g = x-y
\end{sage}
\item Potenz
\begin{sage}
>> f^g
\end{sage}
\begin{sage}
   (x^2 + 3*x + y)^(x - y)
\end{sage}
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Verbinden von Ausdrücken II}
\begin{itemize}
\item Addition / Subtraktion
\begin{sage}
>> f+g, f-g
\end{sage}
\begin{sage}
   (x^2 + 4*x, x^2 + 2*x + 2*y)
\end{sage}
\item Multiplikation/ Division
\begin{sage}
>> f*g, f/g
\end{sage}
\begin{sage}
    ((x - y)*(x^2 + 3*x + y), (x^2 + 3*x + y)/(x - y))
\end{sage}
\end{itemize}
\end{frame}

\begin{frame}[fragile]{collect()}
Durch {\color{blue} \verb+a.collect(Unbestimmte)+} wird der Ausdruck
\verb+a+ bzgl. der \verb+Unbestimmten+ sortiert.
\begin{sage}
>> f = a*x^2+a*x+x^3+sin(x)+b*x+4*x+x*sin(x):
>> f.collect(x)
\end{sage}
\begin{sage}
   a*x^2 + x^3 + (a + b + sin(x) + 4)*x + sin(x)
\end{sage}
\begin{sage}
>> f.collect(x*sin(x))
\end{sage}
\begin{sage}
  a*x^2 + x^3 + a*x + b*x + x*sin(x) + 4*x + sin(x)
\end{sage} 
Durch {\color{blue} \verb+a.combine()+} wird der Ausdruck
durch die Potenzgesetze zusammengefaßt.
\begin{sage}
>> g = x^(a)*x^(b)
>> g.combine()
\end{sage}
\begin{sage}
    x^(a + b)
\end{sage}

\end{frame}

\begin{frame}[fragile]{expand()}
Ausmultiplizieren von  Ausdrücken erfolgt durch {\color{blue}
\verb+a.expand()+} und {\color{blue} \verb+a.expand_trig()+}. \\
\begin{sage}
>> expand((x+2)^4)
\end{sage}
\begin{sage}
 x^4 + 8*x^3 + 24*x^2 + 32*x + 16
\end{sage}
\begin{sage}
>> (sin(x+y)).expand_trig()
\end{sage}
\begin{sage}
  sin(x)*cos(y) + sin(y)*cos(x)
\end{sage}
\end{frame}

\begin{frame}[fragile]{expand() bei Gleichungen}
\begin{sage}
>> a = (16*x-13)^2 == (3*x+5)^2/2
>> a.expand()
\end{sage}
\begin{sage}
  256*x^2 - 416*x + 169 == 9/2*x^2 + 15*x + 25/2
\end{sage}
\begin{sage}
>> a.expand('left')
\end{sage}
\begin{sage}
  256*x^2 - 416*x + 169 == 1/2*(3*x + 5)^2
\end{sage}
\begin{sage}
>> a.expand('right')
\end{sage}
\begin{sage}
  (16*x - 13)^2 == 9/2*x^2 + 15*x + 25/2
\end{sage}
\end{frame}

\begin{frame}[fragile]{factor()}
Der Befehl {\color{blue} \verb+factor(Ausdruck)+} faktorisiert Polynome und
Ausdrücke. 
\begin{itemize}
\item Sage faktorisiert nur, wenn die resultierenden Koeffizienten rationale 
Zahlen sind. 
\item Auch anwendbar auf rationale Funktionen. Es wird ein gemeinsamer
Hauptnenner gesucht.
\end{itemize}
\begin{sage}
>> factor(x^2-2), factor(x^2-9/4)
\end{sage}
\begin{sage}
(x^2 - 2, 1/4*(2*x - 3)*(2*x + 3))
\end{sage}
\begin{sage}
>> factor(2 - 2/(x^2-1))
\end{sage}
\begin{sage}
2*(x^2 - 2)/((x - 1)*(x + 1))
\end{sage}
\end{frame}

\begin{frame}[fragile]{partial\_fraction()}
Durch {\color{blue} \verb+a.partial_fraction()+} wird ein rationaler Ausdruck in eine
Summe rationaler Terme zerlegt, in denen jeweils der Zählergrad
kleiner als der Nennergrad ist. (Partialbruchzerlegung)
\begin{sage}
>> f = x^2/(x^2-1); f.partial_fraction()
\end{sage}
\begin{sage}
 1/2/(x - 1) - 1/2/(x + 1) + 1
\end{sage}
\begin{sage}
f = (x^2+2*x+3)/(x^3+4*x^2+5*x+2); f 
\end{sage}
\begin{sage}
 (x^2 + 2*x + 3)/(x^3 + 4*x^2 + 5*x + 2)
\end{sage}

\end{frame}

\begin{frame}[fragile]{Vergleiche}
\begin{itemize}
\item Der Operator {\color{blue} \verb+==+} vergleicht zwei Objekte. 
\item {\color{blue}
\verb+a==b+} ist wahr (richtig), wenn $a$ und $b$ die gleichen Auswertungen
besitzen (und vom gleichen Typ sind). 
\item Zur Überprüfung von Aussagen gibt es die Funktion
{\color{blue} \verb+bool(Ausdruck)+}. Sie liefert als Ergebnis \verb+True+ oder
\verb+False+.
\item Die inverse Operation zu {\color{blue} \verb+'=='+} ist {\color{blue}
\verb+'<>'+}, also \verb+a<>b+ ist \verb+True+, falls $a$ nicht gleich $b$
ist. 
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Beispiele - Vergleiche I}
\begin{sage}
>> bool(4-3==1)
\end{sage}
\begin{sage}
  True
\end{sage}
\begin{sage}
>> bool(4*x==x); x=0; bool(4*x==x)
\end{sage}
\begin{sage}
 False
 True
\end{sage}
\begin{sage}

\begin{frame}[fragile]{Solve}
\begin{itemize}
\item Solve ist der Befehl zum Lösen von Gleichungen und Gleichungssystemen.
\item Der Befehl ist von der Form {\color{blue} \verb+solve(Gleichungen,Variablen,solution_dict)+}. 
\item {\color{blue}\verb+Gleichungen+} kann ein System von Gleichungen sein.
\item  {\color{blue}\verb+Variablen+} gibt an, wonach aufgelöst wird.
\item Bei einzelnen Gleichungen wird der Lösungswert
zurückgegeben. Bei mehreren Gleichungen wird ein System äquivalenter
Gleichungen zurückgegeben.
\item {\color{blue}\verb+solution_dict=true+} gibt die Lösung als Dictonary zurück.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Beispiele - Solve I} 
\begin{sage}
>> solve(x^2+x == y/4,x)
\end{sage}
\begin{sage}
[x == -1/2*sqrt(y + 1) - 1/2, x == 1/2*sqrt(y + 1) - 1/2]
\end{sage}
%\fcolorbox{red}{blue}{\[\left[x = -\frac{1}{2} \, \sqrt{y + 1} - \frac{1}{2}, x = \frac{1}{2} \,
%\sqrt{y + 1} - \frac{1}{2}\right]\]}
\end{frame}

\begin{frame}[fragile]{Beispiele - Solve II} 
\begin{sage}
>> assume(x>0); solve(x^2+x == y/4,y)
\end{sage}
\begin{sage}
 [y == 4*x^2 + 4*x]
\end{sage}
\begin{sage}
>> solve([x^2-y^2 == 0],[x,y])
\end{sage}
\begin{sage}
([x == -y, x == y], [1, 1])
\end{sage}

>> solve([x^2-y^2 == 0, x+y == 1],x,y)
\end{sage}
\begin{sage}     
  {[x = 1/2, y = 1/2]}
\end{sage}
\end{frame}

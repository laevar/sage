\textbf{Strings}

\begin{itemize}
 \item Deklaration
\begin{sagein}
string = '<Inhalt>'
\end{sagein}
\item Zugriff via Index
\begin{sagein}
string[<Index>]
\end{sagein}
\item Ersetzung
\begin{sagein}
string.replace('<Vorher>','<Hinterher>')
\end{sagein}
\item Zusammenhängen
\begin{sagein}
 string1+string2
\end{sagein}
\item Stringlänge
\begin{sagein}
 len(string)
\end{sagein}
\item Typecast
\begin{sagein}
str(<vorher kein String>)
\end{sagein}
\item Suchen
\begin{sagein}
 string.find(<string>)
\end{sagein}
\item Aufteilen
\begin{sagein}
 string.split(<Trennzeichen>)
\end{sagein}
\item Ausgabe
\begin{sagein}
  print 'Text %<format> und %<format> ... ' % (x,y,...)
 \end{sagein}
\end{itemize}
